üíª Code Example

// Import the Mermaid.js-compatible API from Maid
import { createMermaidAPI } from './maid.bundle.js';
import mermaid from 'mermaid';

// Create Maid renderer instance
const maidRenderer = createMermaidAPI();

// Initialize standard Mermaid.js
mermaid.initialize({ startOnLoad: false });

// Choose renderer dynamically
const useMaid = document.getElementById('useMaidRenderer').checked;
const renderer = useMaid ? maidRenderer : mermaid;

// Use the same API for both!
const { svg } = await renderer.render('diagram-id', diagramText);
document.getElementById('output').innerHTML = svg;
